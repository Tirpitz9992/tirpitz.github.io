如果我们此时是这样的需求：

将x从y的left child转化为parent

那么整棵树的形态应该如何变化呢？

首先要保证，要保证BST的特性，y此时必须变成x的right child

而对于x&y下的其他部分，可以整串subtree同时进行处理

此时显然x原来的right subtree就只能放在y的left subtree

void rratation(int &now)
{
    int l = tree[now].l;
    //left subtree(x)
    tree[now].l = tree[l].r;
    //let x'left subtree become y'right subtree 
    tree[l].r = now;
    now = l;
    //let x become y'parent
    //upd
}
void lroration(int &now)
{
    int r = tree[now].r;
    tree[now].r = tree[r].l;
    tree[r].l = now;
    now = r;
    //upd
}

#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5;
typedef long long ll;
struct Node
{
    int l , r;
    int val , size;
    int cnt; //current node repeat times
}spl[maxn];
int cnt = 0 , root;
void newnode(int &now , int &val)
{
    spl[now = ++cnt].val;
    spl[cnt].size++;
    spl[cnt].cnt ++;
}
void update(int now)
{
    spl[now].size = spl[spl[now].l].size + spl[spl[now].r].size 
    + spl[now].cnt;
    //upd size -> the size of left subtree + the size of right subtree  
    //and + now cnt
}
void rroration(int &now)
{
    int l = spl[now].l;
    spl[now].l = spl[l].r;
    spl[l].r = now;
    now = l;
    update(spl[now].r) , update(now);
}
void lroration(int &now)
{
    int r = spl[now].r;
    spl[now].r= spl[r].l;
    spl[r].l = now;
    now = r;
    update(spl[now].l) , update(now);
}

¶splaying - code

单独考虑当前步骤的状态，判断当前节点为单旋还是双旋

void splaying(int x , int &y)//start x , and end y
{
    if(x == y) return ;
    int &l = spl[y].l , &r = spl[y].r;
    if(x == l) rroration(y);
    else if(x == r) lroration(y);
    else 
    {
        if(spl[x].val < spl[y].val)
        {
            if(spl[x].val < spl[l].val)
                splaying(x , spl[l].l) , rroration(y) , rroration(y);
            else splaying(x , spl[l].r) , lroration(l) , rroration(y);
        }
        else 
        {
            if(spl[x].val > spl[r].val)
                splaying(x , spl[r].r) , lroration(y) , lroration(y);
            else splaying(x , spl[r].l) , rroration(r) , lroration(y);
        }
    }
}

¶insert

插入后，把插入的节点splaying到root node

void insert(int &now , int &val)
{
    if(!now) newnode(now , val) , splaying(now , root);
    else if(val < spl[now].val) insert(spl[now].l , val);
    else if(val > spl[now].val) insert(spl[now].r , val);
    else spl[now].size++ , spl[now].cnt++ , splaying(now , root);
}

¶del

先递归找到需要删除的node ， 然后将node splay到root

如果root.cnt>1，就cnt–

否则，如果root node没有后继(即没有right subtree)，那么就直接让node root变成node root的left child即可(即del此时的root node)

否则，我们寻找root node的后继，然后将其伸展到root node的right child，然后再让root node的left subtree变成root node的right subtree的left subtree即可

void delnode(int now)
{
    splaying(now , root);
    if(spl[now].cnt > 0) spl[now].size-- , spl[now].cnt--;
    else if(spl[root].r)
    {
        int p = spl[root].r;
        while(spl[p].l) p = spl[p].l;
        splaying(p , spl[root].r);
        spl[spl[root].r].l = spl[root].l;
        root = spl[root].r;
        update(root);
    }
    else root = spl[root].l;
}

//

void del(int now , int &val)
{
    if(spl[now].val == val) delnode(now);
    else if(val < spl[now].val) del(spl[now].l , val);
    else del(spl[now].r , val);
}

int getrank(int val)
{
    int now = root, rank = 1;

    {
        if(spl[now].val==val)   //找到了要的结点，这个之前的没有

    return rank;

¶区间操作

如果要对$[L , R]$区间操作

那么我们把$L - 1$这个节点伸展到root node,再把$R + 1$这个节点伸展到root node的right child，那么此时node root的right child的left subtree就是此时要操作的区间

需要注意的是，在具体问题中，我们需要插入一个INF和一个-INF去避免$L - 1$为零的问题
